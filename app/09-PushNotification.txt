{/**react-native-push-notification não funciona com o Expo Go.**/}

import React, { useEffect } from 'react';
import { Button, View } from 'react-native';
import PushNotification from 'react-native-push-notification';

export default function PushNotificationExample() {
  useEffect(() => {

    PushNotification.configure({
      onNotification: function (notification) {
        console.log('Notificação recebida:', notification);
      },
      requestPermissions: true,
    });
  }, []);

  const scheduleNotification = () => {
    PushNotification.localNotificationSchedule({
      message: 'Esta é uma notificação agendada (RNP)',
      date: new Date(Date.now() + 5 * 1000), 
      allowWhileIdle: true,
    });
  };

  return (
    <View style={{ padding: 20 }}>
      <Button title="Agendar Notificação (RNP)" onPress={scheduleNotification} />
    </View>
  );
}
